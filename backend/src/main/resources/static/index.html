Supabase Database Setup

Connect your Supabase database for persistent appointment, patient and report storage. Free tier available at supabase.com

Run this SQL in Supabase → SQL Editor: 
-- Appointments table
CREATE TABLE IF NOT EXISTS appointments (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL, phone TEXT NOT NULL, email TEXT,
  service TEXT, preferred_date DATE, message TEXT,
  status TEXT DEFAULT 'pending', notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Patients table
CREATE TABLE IF NOT EXISTS patients (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL, phone TEXT NOT NULL, email TEXT,
  dob DATE, blood_group TEXT, address TEXT,
  allergies TEXT, medical_history TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Reports table
CREATE TABLE IF NOT EXISTS reports (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  patient_id UUID REFERENCES patients(id) ON DELETE CASCADE,
  title TEXT NOT NULL, description TEXT,
  file_url TEXT, file_name TEXT, file_type TEXT,
  report_date DATE DEFAULT CURRENT_DATE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Storage bucket (run separately or create in Supabase dashboard)
-- INSERT INTO storage.buckets (id, name, public) VALUES ('reports', 'reports', false);

-- Row Level Security
ALTER TABLE appointments ENABLE ROW LEVEL SECURITY;
ALTER TABLE patients ENABLE ROW LEVEL SECURITY;
ALTER TABLE reports ENABLE ROW LEVEL SECURITY;

CREATE POLICY "public_book" ON appointments FOR INSERT TO anon WITH CHECK (true);
CREATE POLICY "admin_appts" ON appointments FOR ALL TO authenticated USING (true);
CREATE POLICY "admin_patients" ON patients FOR ALL TO authenticated USING (true);
CREATE POLICY "admin_reports" ON reports FOR ALL TO authenticated USING (true);
